  <script>
    function showSection(id, push=true) {
      document.querySelectorAll("section").forEach(s => s.style.display = "none");
      document.getElementById(id).style.display = "block";
      if (id === "home") document.getElementById("offers").style.display = "block";
      if (push) history.pushState({ section:id }, "", "#" + id);
      window.scrollTo({ top: 0, behavior: "smooth" });
    }

    window.onpopstate = e => {
      const sec = e.state?.section || "home";
      showSection(sec, false);
    };

    window.onload = () => {
      const hash = location.hash.replace("#", "") || "home";
      showSection(hash, false);
    };

  // Submission guidelines logic
  const form = document.querySelector('#submit form');
  const guidelinesModal = document.getElementById('guidelinesModal');
  let formShouldSubmit = false;

  if (form) {
    form.addEventListener('submit', function(e) {
      if (!formShouldSubmit) {
        e.preventDefault();
        // Check if form is valid
        if (form.checkValidity()) {
          guidelinesModal.style.display = 'flex';
        } else {
          form.reportValidity();
        }
      }
    });
  }

  document.getElementById('confirmSubmit').onclick = function() {
    formShouldSubmit = true;
    guidelinesModal.style.display = 'none';
    form.submit();
  };

  function closeGuidelines() {
    guidelinesModal.style.display = 'none';
  }
  </script>
